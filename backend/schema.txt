generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id          String    @id @default(cuid())
  email       String    @unique
  password    String
  name        String    @unique
  role        Role      @default(STAFF)
  isActive    Boolean   @default(true)
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt
  permissions Json?     @default("{\"tyres\": true, \"users\": false, \"history\": true, \"stepney\": true, \"vehicles\": true, \"allotment\": true, \"dashboard\": true, \"documents\": true, \"locations\": true}")
  history     History[]

  @@map("users")
}

model Make {
  id        String    @id @default(cuid())
  name      String    @unique
  createdAt DateTime  @default(now())
  vehicles  Vehicle[]

  @@map("makes")
}

model VehicleType {
  id        String    @id @default(cuid())
  name      String    @unique
  createdAt DateTime  @default(now())
  vehicles  Vehicle[]

  @@map("vehicle_types")
}

model Vehicle {
  id           String            @id @default(cuid())
  reg          String            @unique
  type         String
  model        String
  stepneySlots Int               @default(1)
  createdAt    DateTime          @default(now())
  updatedAt    DateTime          @updatedAt
  makeId       String?
  typeId       String?
  axles        Axle[]
  challans     Challan[]
  gps_devices  gps_devices[]
  history      History[]
  stepneys     Tyre[]            @relation("VehicleStepneys")
  tyres        Tyre[]
  documents    VehicleDocument[]
  make         Make?             @relation(fields: [makeId], references: [id])
  vehicleType  VehicleType?      @relation(fields: [typeId], references: [id])

  @@map("vehicles")
}

model Axle {
  id        String  @id @default(cuid())
  vehicleId String
  tyreCount Int
  steering  Boolean @default(false)
  drive     Boolean @default(false)
  line      Int?
  sortOrder Int     @default(0)
  vehicle   Vehicle @relation(fields: [vehicleId], references: [id], onDelete: Cascade)
  tyres     Tyre[]

  @@map("axles")
}

model Tyre {
  id               String       @id @default(cuid())
  serial           String       @unique
  brand            String
  size             String
  pattern          String?
  purchaseDate     DateTime?
  initialTread     Float        @default(20)
  currentTread     Float        @default(20)
  status           TyreStatus   @default(INVENTORY)
  cost             Float?
  remarks          String?
  withRim          Boolean      @default(true)
  vehicleId        String?
  axleId           String?
  position         String?
  mountingDate     DateTime?
  stepneyVehicleId String?
  stepneyDate      DateTime?
  stepneyType      StepneyType?
  locationId       String?
  createdAt        DateTime     @default(now())
  updatedAt        DateTime     @updatedAt
  history          History[]
  images           TyreImage[]
  axle             Axle?        @relation(fields: [axleId], references: [id])
  location         Location?    @relation(fields: [locationId], references: [id])
  stepneyVehicle   Vehicle?     @relation("VehicleStepneys", fields: [stepneyVehicleId], references: [id])
  vehicle          Vehicle?     @relation(fields: [vehicleId], references: [id])

  @@map("tyres")
}

model TyreImage {
  id        String    @id @default(cuid())
  tyreId    String
  url       String
  type      ImageType @default(TYRE)
  caption   String?
  createdAt DateTime  @default(now())
  tyre      Tyre      @relation(fields: [tyreId], references: [id], onDelete: Cascade)

  @@map("tyre_images")
}

model Location {
  id        String       @id @default(cuid())
  name      String
  type      LocationType
  address   String?
  contact   String?
  createdAt DateTime     @default(now())
  updatedAt DateTime     @updatedAt
  tyres     Tyre[]

  @@map("locations")
}

model History {
  id        String   @id @default(cuid())
  date      DateTime @default(now())
  action    String
  details   String?
  tyreId    String
  vehicleId String?
  axleIndex Int?
  position  String?
  userId    String?
  createdAt DateTime @default(now())
  tyre      Tyre     @relation(fields: [tyreId], references: [id], onDelete: Cascade)
  user      User?    @relation(fields: [userId], references: [id])
  vehicle   Vehicle? @relation(fields: [vehicleId], references: [id])

  @@map("history")
}

model Challan {

  vehicleId         String
  challanNumber     String           @unique
  challanDate       DateTime
  offence           String
  section           String?
  fineAmount        Float
  penaltyAmount     Float?
  totalAmount       Float

  paymentReceipt    String?

  paymentGatewayRef String?
  location          String?
  evidenceUrl       String?

  createdAt         DateTime         @default(now())

  payments          ChallanPayment[]

  @@map("challans")
}

  method        PaymentMethod
  gatewayRef    String?

  initiatedAt   DateTime      @default(now())
  completedAt   DateTime?
  failureReason String?
  receiptUrl    String?
  challan       Challan       @relation(fields: [challanId], references: [id], onDelete: Cascade)

  @@map("challan_payments")
}

model VehicleDocument {
  id         String             @id @default(cuid())
  vehicleId  String
  type       DocumentType
  number     String?
  issuer     String?
  issueDate  DateTime?
  expiryDate DateTime?
  amount     Float?
  status     DocumentStatus     @default(ACTIVE)
  fileUrl    String?
  notes      String?
  createdAt  DateTime           @default(now())
  updatedAt  DateTime           @updatedAt
  reminders  DocumentReminder[]
  vehicle    Vehicle            @relation(fields: [vehicleId], references: [id], onDelete: Cascade)

  @@map("vehicle_documents")
}

model DocumentReminder {
  id         String          @id @default(cuid())
  documentId String
  type       ReminderType
  daysBefore Int
  sentAt     DateTime?
  message    String?
  createdAt  DateTime        @default(now())
  document   VehicleDocument @relation(fields: [documentId], references: [id], onDelete: Cascade)

  @@index([documentId])
  @@index([sentAt])
  @@map("document_reminders")
}

model gps_alerts {
  id              Int          @id @default(autoincrement())
  device_id       Int?
  alert_type      String       @db.VarChar(50)
  severity        String?      @default("MEDIUM") @db.VarChar(20)
  message         String?
  latitude        Decimal?     @db.Decimal(10, 8)
  longitude       Decimal?     @db.Decimal(11, 8)
  acknowledged    Boolean?     @default(false)
  acknowledged_at DateTime?    @db.Timestamptz(6)
  created_at      DateTime?    @default(now()) @db.Timestamptz(6)
  gps_devices     gps_devices? @relation(fields: [device_id], references: [id], onDelete: Cascade, onUpdate: NoAction)

  @@index([device_id, created_at(sort: Desc)], map: "idx_gps_alerts_device")
  @@index([acknowledged, created_at(sort: Desc)], map: "idx_gps_alerts_unack")
}

model gps_devices {
  id                     Int             @id @default(autoincrement())
  traccar_device_id      Int             @unique
  vehicle_id             String?
  current_vehicle_reg    String?         @db.VarChar(50)
  current_vehicle_make   String?         @db.VarChar(50)
  current_vehicle_model  String?         @db.VarChar(50)
  current_vehicle_year   Int?
  assigned_at            DateTime?       @db.Timestamptz(6)
  device_name            String?         @db.VarChar(100)
  imei                   String?         @db.VarChar(50)
  sim_number             String?         @db.VarChar(20)
  status                 String?         @default("OFFLINE") @db.VarChar(20)
  latitude               Decimal?        @db.Decimal(10, 8)
  longitude              Decimal?        @db.Decimal(11, 8)
  speed                  Decimal?        @db.Decimal(5, 2)
  heading                Decimal?        @db.Decimal(5, 2)
  ignition               Boolean?        @default(false)
  battery_level          Int?
  geofence_enabled       Boolean?        @default(false)
  geofence_center_lat    Decimal?        @db.Decimal(10, 8)
  geofence_center_lng    Decimal?        @db.Decimal(11, 8)
  geofence_radius        Int?            @default(500)
  last_ping              DateTime?       @db.Timestamptz(6)
  created_at             DateTime?       @default(now()) @db.Timestamptz(6)
  updated_at             DateTime?       @default(now()) @db.Timestamptz(6)
  gps_alerts             gps_alerts[]
  vehicles               Vehicle?        @relation(fields: [vehicle_id], references: [id], onUpdate: NoAction)
  gps_geofences          gps_geofences[]
  gps_positions          gps_positions[]
  gps_trips              gps_trips[]

  @@index([traccar_device_id], map: "idx_gps_devices_traccar")
  @@index([vehicle_id], map: "idx_gps_devices_vehicle")
}

model gps_geofences {
  id          Int          @id @default(autoincrement())
  device_id   Int?
  name        String?      @db.VarChar(100)
  center_lat  Decimal      @db.Decimal(10, 8)
  center_lng  Decimal      @db.Decimal(11, 8)
  radius      Int
  is_active   Boolean?     @default(true)
  created_at  DateTime?    @default(now()) @db.Timestamptz(6)
  gps_devices gps_devices? @relation(fields: [device_id], references: [id], onDelete: Cascade, onUpdate: NoAction)
}

model gps_positions {
  id                  Int          @id @default(autoincrement())
  device_id           Int?
  traccar_position_id BigInt?
  latitude            Decimal      @db.Decimal(10, 8)
  longitude           Decimal      @db.Decimal(11, 8)
  altitude            Decimal?     @db.Decimal(8, 2)
  speed               Decimal?     @db.Decimal(5, 2)
  heading             Decimal?     @db.Decimal(5, 2)
  accuracy            Decimal?     @db.Decimal(5, 2)
  ignition            Boolean?
  battery_level       Int?
  recorded_at         DateTime     @db.Timestamptz(6)
  created_at          DateTime?    @default(now()) @db.Timestamptz(6)
  gps_devices         gps_devices? @relation(fields: [device_id], references: [id], onDelete: Cascade, onUpdate: NoAction)

  @@index([device_id, recorded_at(sort: Desc)], map: "idx_gps_positions_device_time")
}

model gps_trips {
  id          Int          @id @default(autoincrement())
  device_id   Int?
  start_lat   Decimal?     @db.Decimal(10, 8)
  start_lng   Decimal?     @db.Decimal(11, 8)
  end_lat     Decimal?     @db.Decimal(10, 8)
  end_lng     Decimal?     @db.Decimal(11, 8)
  start_time  DateTime?    @db.Timestamptz(6)
  end_time    DateTime?    @db.Timestamptz(6)
  distance    Decimal?     @db.Decimal(10, 2)
  duration    Int?
  max_speed   Decimal?     @db.Decimal(5, 2)
  avg_speed   Decimal?     @db.Decimal(5, 2)
  created_at  DateTime?    @default(now()) @db.Timestamptz(6)
  gps_devices gps_devices? @relation(fields: [device_id], references: [id], onDelete: Cascade, onUpdate: NoAction)
}
